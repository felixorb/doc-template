% Double column header - 双列 Header 样式：| 姓名 | 邮箱/主页/电话等（这里可以调整展示项的个数）|
\newcommand{\doublecol}[4]{
  \begin{tabularx}{\textwidth}{Xr}
    {
      \begin{tabular}[c]{l}
        % \fontsize{35}{45}\selectfont{\color{primary}{{\textbf{\fullname}}}} \\
        % {\textit{\subtitle}} % You could add a subtitle here
        \hspace{1em}%
        \fontsize{25}{30}\selectfont{\color{primary}{{\textbf{\fullname}}}}
      \end{tabular}
    } & {
      \begin{tabular}[c]{l@{\hspace{1.5em}}l}
        {\small#1} & {\small#2} \\
        {\small#3} & {\small#4} \\
      \end{tabular}
    }
  \end{tabularx}
}

% Double column header with photo  - 双列 Header 样式：| 头像+姓名 | 邮箱/主页/电话等（这里可以调整展示项的个数）|
\newcommand{\doublecolwithphoto}[4]{
  \hspace{1em}%
  \parbox{2cm}{
    \includegraphics[width=2cm,height=2.8cm,clip]{\avatar}
  }
  \hspace{1em}%
  \begin{tabularx}{0.8\textwidth}{Xr}
    {
      \begin{tabular}[c]{l}
        % \hspace{1em}%
        {\fontsize{20}{30}\selectfont{\color{primary}{{\textbf{\fullname}}}}} \\
        {\textit{\gender}} \\% You could add a subtitle here
        {\textit{\birthdate}} % You could add a subtitle here
      \end{tabular}
    } & {
      \begin{tabular}[c]{l@{\hspace{1.5em}}l}
        {\small#1} & {\small#2} \\
        {\small#3} & {\small#4} \\
      \end{tabular}
    }
  \end{tabularx}
}
